// 实现HMAC-SHA256签名
func (s *NotificationService) generateSignature(data []byte, secret string) string {
    mac := hmac.New(sha256.New, []byte(secret))
    mac.Write(data)
    return hex.EncodeToString(mac.Sum(nil))
}

// 添加时间戳防止重放攻击
